function getNextFridayMidnightKyiv() { const now = new Date(); // Get the current date in Kyiv timezone const formatter = new Intl.DateTimeFormat('en-US', { timeZone: 'Europe/Kyiv', weekday: 'short', year: 'numeric', month: 'numeric', day: 'numeric', hour: 'numeric', minute: 'numeric', second: 'numeric', hour12: false, }); const parts = formatter.formatToParts(now).reduce((acc, part) => { acc[part.type] = part.value; return acc; }, {}); const currentKyiv = { year: +parts.year, month: +parts.month, day: +parts.day, hour: +parts.hour, minute: +parts.minute, second: +parts.second, weekday: parts.weekday, }; const daysOfWeek = { Sun: 0, Mon: 1, Tue: 2, Wed: 3, Thu: 4, Fri: 5, Sat: 6 }; const currentDay = daysOfWeek[currentKyiv.weekday]; let daysUntilFriday = (5 - currentDay + 7) % 7; // If it's Friday and past midnight in Kyiv, move to next Friday if (daysUntilFriday === 0 && (currentKyiv.hour > 0 || currentKyiv.minute > 0 || currentKyiv.second > 0)) { daysUntilFriday = 7; } // Target Kyiv Friday at 00:00 const targetKyivDate = new Date(Date.UTC( currentKyiv.year, currentKyiv.month - 1, currentKyiv.day + daysUntilFriday, 0, 0, 0 )); // Convert Kyiv midnight to UTC const utcOffsetMs = new Date(targetKyivDate.toLocaleString("en-US", { timeZone: "Europe/Kyiv" })).getTime() - targetKyivDate.getTime(); return new Date(targetKyivDate.getTime() - utcOffsetMs); }
